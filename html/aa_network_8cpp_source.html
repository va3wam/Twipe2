64 const char* aaNetwork::_lookForAP()
65 {
66  Serial.println("<aaNetwork::_lookForAP> Scanning the 2.4GHz radio spectrum for known Access Points.");
67  _ssid = _unknownAP; // At the start no known Access Point has been foundto connect to
68  int numberOfNetworks = WiFi.scanNetworks(); // Used to track how many APs are detected by the scan
69  int StrongestSignal = -127; // Used to find the strongest signal. Set as low as possible to start
70  bool APknown; // Flag to indicate if the current AP appears in the known AP list
71  Serial.println(numberOfNetworks);
72 
73  // Loop through all detected APs
74  for(int i = 0; i < numberOfNetworks; i++)
75  {
76  APknown = false;
77 
78  // Scan table of known APs to see if the current AP is known to us
79  for (int j = 0; j < numKnownAPs; j++)
80  {
81  // If the current scanned AP appears in the known AP list note the index value and flag found
82  if(WiFi.SSID(i) == SSID[j])
83  {
84  APknown = true;
85  _SSIDIndex = j;
86  } //if
87  } //for
88 
89  // If the current AP is known and has a stronger signal than the others that have been checked
90  // then store it in the variables that will be used to connect to the AP later
91  if((APknown == true) && (WiFi.SSID(i).toInt() > StrongestSignal))
92  {
93  _ssid = SSID[_SSIDIndex].c_str();
94  _password = Password[_SSIDIndex].c_str();
95  StrongestSignal = WiFi.SSID(i).toInt();
96  } //if
97  } //for
98  return _ssid;
99 } //aaNetwork::lookForAP()
