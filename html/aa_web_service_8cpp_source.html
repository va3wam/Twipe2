245 void aaWebService::_cfgOtaPageHandler()
246 {
247  /*handling uploading firmware file */
248  server.on("/update", HTTP_POST, []()
249  {
250  server.sendHeader("Connection", "close");
251  server.send(200, "text/plain", (Update.hasError()) ? "FAIL" : "OK");
252  ESP.restart();
253  }, // server.on("/update")
254  []()
255  {
256  HTTPUpload& upload = server.upload();
257  if(upload.status == UPLOAD_FILE_START)
258  {
259  Serial.printf("Update: %s\n", upload.filename.c_str());
260  if(!Update.begin(UPDATE_SIZE_UNKNOWN))
261  { //start with max available size
262  Update.printError(Serial);
263  } //if
264  } //if
265  else if(upload.status == UPLOAD_FILE_WRITE)
266  {
267  /* flashing firmware to ESP*/
268  if (Update.write(upload.buf, upload.currentSize) != upload.currentSize)
269  {
270  Update.printError(Serial);
271  } //if
272  } //else if
273  else if (upload.status == UPLOAD_FILE_END)
274  {
275  if (Update.end(true)) //true to set the size to the current progress
276  {
277  Serial.printf("Update Success: %u\nRebooting...\n", upload.totalSize);
278  } // if
279  else
280  {
281  Update.printError(Serial);
282  } //else
283  } //else if
284  }); // []()
285 } //aaWebService::_cfgOtaPageHandler()

318 bool aaWebService::newMqttBrokerIp(const char* address) // Handle new IP address for broker from web.
319 {
320  int8_t numPings = 1; // How many pings to send to verify IP address
321  IPAddress tmpIp;
322  tmpIp.fromString(address);
323  bool ret = Ping.ping(address, numPings);
324  Serial.print("<aaWebService::newMqttBrokerIp> Sent "); Serial.print(numPings);
325  Serial.print(" pings to new IP address "); Serial.print(tmpIp);
326  Serial.print(" and got the response "); Serial.println(ret);
327  if(ret == 0)
328  {
329  Serial.println("<aaWebService::newMqttBrokerIp> New broker IP address is invalid and has been discarded.");
330  optionMessage = "Broker IP rejected. Keeping Old Address";
331  } //if
332  else
333  {
334  Serial.print("<aaWebService::newMqttBrokerIp> MQTT broker IP will change to "); Serial.println(tmpIp);
335  optionMessage = "Broker IP successfully updated";
336  newBrokerIp = tmpIp; // Store new broker IP to be collected by main later.
337  haveNewBrokerIp = true; // Signal that new broker IP is available.
338  } //else
339  return ret;
340 } //aaWebService::newMqttBrokerIp()
