59 void aaMqtt::connect(IPAddress address, const char* uniqueName)
60 {
61  _brokerIP = address;
62  _uniqueName = uniqueName; // Prefix for all MQTT topic tree names.
63  Serial.print("<aaMqtt::connect> Connecting as ");
64  Serial.print(_uniqueName);
65  Serial.print(" to MQTT broker at IP address ");
66  Serial.print(_brokerIP);
67  Serial.println(" ...");
68  mqttClient.onConnect(onMqttConnect); // Define connection event.
69  mqttClient.onDisconnect(onMqttDisconnect); // Define disconnect event.
70  mqttClient.onSubscribe(onMqttSubscribe); // Define subscribe event.
71  mqttClient.onUnsubscribe(onMqttUnsubscribe); // Define unsubscribe event.
72  mqttClient.onMessage(onMqttMessage); // Define recieve message event.
73  mqttClient.onPublish(onMqttPublish); // Define publish message event.
74  mqttClient.setServer(_brokerIP, BROKER_PORT); // Set IP and port for broker service.
75  mqttClient.connect(); // Connect to broker.
76 } //aaMqtt::connect()
