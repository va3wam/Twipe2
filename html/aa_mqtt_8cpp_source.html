84  mqttClient.onConnect(onMqttConnect); // Define connection event.
85  mqttClient.onDisconnect(onMqttDisconnect); // Define disconnect event.
86  mqttClient.onSubscribe(onMqttSubscribe); // Define subscribe event.
87  mqttClient.onUnsubscribe(onMqttUnsubscribe); // Define unsubscribe event.
88  mqttClient.onMessage(onMqttMessage); // Define recieve message event.
89  mqttClient.onPublish(onMqttPublish); // Define publish message event.
90  mqttClient.setServer(_brokerIP, BROKER_PORT); // Set IP and port for broker service.
91  mqttClient.connect(); // Connect to broker.
